Implementing Countdown problem with Scala

Haskell was born functional. I find that functional programming in Haskell looks more succint than in Scala, which is also functional langauge. But Scala was born on top of java to extend functional features. So, Scala is more like hybrid language of object oriented and functioanl features. It means that Scala has OOP, functional, strict, lazy, mutable, immutable, so many features. It all depends on how scala users utilize it. So Scala looks a bit complicated for starters.
Nevertheless, Scala can be used as much similar as Haskell for functional programming. Here, I translate Haskell code of Countdown problem from Graham Hutton’s book Programming in Haskell into Scala language.

The Haskell version of the code:

-- Countdown example from chapter 9 of Programming in Haskell,
-- Graham Hutton, Cambridge University Press, 2016.

-- Arithmetic operators

data Op = Add | Sub | Mul | Div

instance Show Op where
   show Add = "+"
   show Sub = "-"
   show Mul = "*"
   show Div = "/"

valid :: Op -> Int -> Int -> Bool
valid Add _ _ = True
valid Sub x y = x > y
valid Mul _ _ = True
valid Div x y = x `mod` y == 0

apply :: Op -> Int -> Int -> Int
apply Add x y = x + y
apply Sub x y = x - y
apply Mul x y = x * y
apply Div x y = x `div` y

-- Numeric expressions

data Expr = Val Int | App Op Expr Expr

instance Show Expr where
   show (Val n)     = show n
   show (App o l r) = brak l ++ show o ++ brak r
                      where
                         brak (Val n) = show n
                         brak e       = "(" ++ show e ++ ")"

values :: Expr -> [Int]
values (Val n)     = [n]
values (App _ l r) = values l ++ values r

eval :: Expr -> [Int]
eval (Val n)     = [n | n > 0]
eval (App o l r) = [apply o x y | x <- eval l,
                                  y <- eval r,
                                  valid o x y]

-- Combinatorial functions

subs :: [a] -> [[a]]
subs []     = [[]]
subs (x:xs) = yss ++ map (x:) yss
              where yss = subs xs

interleave :: a -> [a] -> [[a]]
interleave x []     = [[x]]
interleave x (y:ys) = (x:y:ys) : map (y:) (interleave x ys)

perms :: [a] -> [[a]]
perms []     = [[]]
perms (x:xs) = concat (map (interleave x) (perms xs))

choices :: [a] -> [[a]]
choices = concat . map perms . subs

-- Formalising the problem

solution :: Expr -> [Int] -> Int -> Bool
solution e ns n = elem (values e) (choices ns) && eval e == [n]

-- Brute force solution

split :: [a] -> [([a],[a])]
split []     = []
split [_]    = []
split (x:xs) = ([x],xs) : [(x:ls,rs) | (ls,rs) <- split xs]

exprs :: [Int] -> [Expr]
exprs []  = []
exprs [n] = [Val n]
exprs ns  = [e | (ls,rs) <- split ns,
                 l       <- exprs ls,
                 r       <- exprs rs,
                 e       <- combine l r]

combine :: Expr -> Expr -> [Expr]
combine l r = [App o l r | o <- ops]

ops :: [Op]
ops = [Add,Sub,Mul,Div]

solutions :: [Int] -> Int -> [Expr]
solutions ns n = [e | ns' <- choices ns, e <- exprs ns', eval e == [n]]

-- Combining generation and evaluation

type Result = (Expr,Int)

results :: [Int] -> [Result]
results []  = []
results [n] = [(Val n,n) | n > 0]
results ns  = [res | (ls,rs) <- split ns,
                      lx     <- results ls,
                      ry     <- results rs,
                      res    <- combine' lx ry]

combine' :: Result -> Result -> [Result]
combine' (l,x) (r,y) = [(App o l r, apply o x y) | o <- ops, valid o x y]

solutions' :: [Int] -> Int -> [Expr]
solutions' ns n = [e | ns' <- choices ns, (e,m) <- results ns', m == n]

-- Exploiting algebraic properties

valid' :: Op -> Int -> Int -> Bool
valid' Add x y = x <= y
valid' Sub x y = x > y
valid' Mul x y = x /= 1 && y /= 1 && x <= y
valid' Div x y = y /= 1 && x `mod` y == 0

results' :: [Int] -> [Result]
results' []  = []
results' [n] = [(Val n,n) | n > 0]
results' ns  = [res | (ls,rs) <- split ns,
                       lx     <- results' ls,
                       ry     <- results' rs,
                       res    <- combine'' lx ry]

combine'' :: Result -> Result -> [Result]
combine'' (l,x) (r,y) = [(App o l r, apply o x y) | o <- ops, valid' o x y]

solutions'' :: [Int] -> Int -> [Expr]
solutions'' ns n = [e | ns' <- choices ns, (e,m) <- results' ns', m == n]

-- Testing
main :: IO ()
main = print (solutions'' [3,5,7,12,18,20] 750)
-- main = print (solutions'' [1,3,7,10,25,50] 765)

Here is Graham Hutton’s video lecture explaining countdown problem in Haskell language.

The Haskell code above can be implemented with Scala language in very much similar functional style, Here is Scala implementation. We need to use Scala’s OO features(trait, class, object, inheritance) to construct algebraic data type. First, algebraic data type implementation with trait and case object in Scala:

trait Op {
  def apply(a: Int, b: Int): Int
}

case object Add extends Op {
  def apply(a: Int, b: Int) = a + b
}
case object Sub extends Op {
  def apply(a: Int, b: Int) = a - b
}
case object Mul extends Op {
  def apply(a: Int, b: Int) = a * b
}
case object Div extends Op {
  def apply(a: Int, b: Int) = a / b
}

trait Expr {
  override def toString = this match {
    case Val(v) => v.toString
    case Aps(o, l, r) => o match {
      case Add => "(" + l.toString + " + " + r.toString + ")"
      case Sub => "(" + l.toString + " - " + r.toString + ")"
      case Mul => "(" + l.toString + " * " + r.toString + ")"
      case Div => "(" + l.toString + " / " + r.toString + ")"
    }
  }
}

case class Val(value: Int) extends Expr
case class Aps(o: Op, l: Expr, r: Expr) extends Expr

With Op, Expr, Val, Aps types at hand, we can implement Countdown object and functions inside it to solve countdown problem. Alternative solution functions included.

object CountDown extends App {

  def valid(op: Op, x: Int, y: Int): Boolean = op match {
    case Add => true
    case Sub => x > y
    case Mul => true
    case Div => (x % y == 0)
  }

  def values(e: Expr): List[Int] = e match {
    case Val(n) => List(n)
    case Aps(_, l, r) => values(l) ++ values(r)
  }

  //use List instead of Option
  def eval(e: Expr): List[Int] = e match {
    case Val(n) => if (n > 0) List(n) else Nil
    case Aps(o, l, r) => for {
      x <- eval(l)
      y <- eval(r) if valid(o, x, y) //else eval gets Nil
    } yield o(x, y)
  }

  //combinatorial FNs
  def subs[T](xs: List[T]): List[List[T]] = xs match {
    case Nil => List(Nil)
    case a :: as =>
      val ys = subs(as)
      ys ++ ys.map(a :: _)
  }
  def interleave[T](x: T, xs: List[T]): List[List[T]] = xs match {
    case Nil => List(List(x))
    case y :: ys => (x :: xs) :: (interleave(x, ys).map(y :: _))
  }
  def perms[T](xs: List[T]): List[List[T]] = xs match {
    case Nil => List(Nil)
    case y :: ys => perms(ys).flatMap(interleave(y, _))
  }
  def choices[T](xs: List[T]): List[List[T]] = for {
    ys <- subs(xs)
    zs <- perms(ys)
  } yield zs

  //solution e for ns, n is correct?
  def solution(e: Expr, ns: List[Int], n: Int): Boolean =
    choices(ns).contains(values(e)) && eval(e) == List(n)

  //Brute force solution
  def split(xs: List[Int]): List[(List[Int], List[Int])] = xs match {
    case Nil => Nil
    case a :: Nil => Nil
    case a :: as => (List(a), as) :: (for {
      (ls, rs) <- split(as)
    } yield (a :: ls, rs))
  }
  val ops = List(Add, Sub, Mul, Div)

  def combine(l: Expr, r: Expr): List[Expr] = ops.map(Aps(_, l, r))

  def exprs(ns: List[Int]): List[Expr] = ns match {
    case Nil => Nil
    case a :: Nil => List(Val(a))
    case _ => for {
      (ls, rs) <- split(ns)
      l <- exprs(ls)
      r <- exprs(rs)
      e <- combine(l, r)
    } yield e
  }

  def solutions(ns: List[Int], n: Int): List[Expr] = for {
    ls <- choices(ns)
    e <- exprs(ls) if eval(e) == List(n)
  } yield e

  //Combining generation and evaluation
  type Result = (Expr, Int)

  def combine2(r1: Result, r2: Result): List[Result] = {
    val (l, x) = r1
    val (r, y) = r2
    ops.filter(valid(_, x, y)).map(o => (Aps(o, l, r), o(x, y)))
  }

  def results(is: List[Int]): List[Result] = is match {
    case Nil => Nil
    case n :: Nil => if (n > 0) List((Val(n), n)) else Nil
    case ns => for {
      (ls, rs) <- split(ns)
      lx <- results(ls)
      ry <- results(rs)
      res <- combine2(lx, ry)
    } yield res
  }

  def solutions2(ns: List[Int], n: Int): List[Expr] = for {
    xs <- choices(ns)
    (e, m) <- results(xs) if m == n
  } yield e


  //Exploiting numeric properties
  def valid2(op: Op, x: Int, y: Int): Boolean = op match {
    case Add => x <= y
    case Sub => x > y
    case Mul => x != 1 && y != 1 && x <= y
    case Div => y != 1 && (x % y == 0)
  }

  def combine3(r1: Result, r2: Result): List[Result] = {
    val (l, x) = r1
    val (r, y) = r2
    ops.filter(valid2(_, x, y)).map(o => (Aps(o, l, r), o(x, y)))
  }

  def results2(is: List[Int]): List[Result] = is match {
    case Nil => Nil
    case n :: Nil => if (n > 0) List((Val(n), n)) else Nil
    case ns => for {
      (ls, rs) <- split(ns)
      lx <- results2(ls)
      ry <- results2(rs)
      res <- combine3(lx, ry)
    } yield res
  }

  def solutions3(ns: List[Int], n: Int): List[Expr] = for {
    xs <- choices(ns)
    (e, m) <- results2(xs) if m == n
  } yield e

  val result = solutions3(List(3,5,7,12,18,20), 750)

  println(result.length)

  result foreach {
    e => println(e)
  }
}

Both Haskell and Scala implementations computes the same result.
